```python
from pathlib import Path
from app.context import CodeParserService, ContextRetriever, language_for_path

def test_language_registry_covers_mainstream_languages():
    paths = ["a.ts", "b.py", "c.java", "d.go", "e.rs", "f.cpp", "g.cs", "h.php", "i.rb", "j.kt", "k.swift", "l.dart", "m.scala"]
    assert {language_for_path(path).language_id for path in paths} >= {"typescript", "python", "java", "go", "rust", "cpp", "csharp", "php", "ruby", "kotlin", "swift", "dart", "scala"}

def test_parser_extracts_imports_functions_and_classes():
    parser = CodeParserService()
    result = parser.parse_text("sample.py", "import os\nclass Service:\n    def run(self):\n        return os.getcwd()\n")
    assert result.language == "python"
    assert "os" in result.imports
    assert "Service" in result.classes
    assert "run" in result.functions

def test_context_retriever_is_bounded_and_prefers_relevant_files(tmp_path: Path):
    (tmp_path / "payment.py").write_text("class PaymentService:\n    def refund(self):\n        pass\n", encoding="utf-8")
    (tmp_path / "unrelated.py").write_text("def paint():\n    pass\n", encoding="utf-8")
    result = ContextRetriever(max_files=1).retrieve(tmp_path, "fix payment refund")
    assert [item.path for item in result.items] == ["payment.py"]
    assert result.scanned_file_count == 2
    assert result.truncated is True
```
